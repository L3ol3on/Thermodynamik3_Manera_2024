\subsection{Forced and Free convection}
    \begin{minipage}{0.49\linewidth}
        \titel{forced}
        \vspace*{-2em}
        \begin{align*}
            Nu = f(Re, Pr)\\
            \text{Average Nusselt number:} \quad Nu = C \cdot Re^n \cdot Pr^{\frac{1}{3}}
        \end{align*}
    \end{minipage}
    \begin{minipage}{0.49\linewidth}
        \titel{free}
        \vspace*{-2em}
        \begin{align*}
            Nu = f(Gr, Pr)\\
            \text{Average Nusselt number:} \quad Nu = C \cdot Gr^m \cdot Pr^m
        \end{align*}
    \end{minipage}
