\subsection{Dependence of temperature on time}
\begin{minipage}{0.54\linewidth}
    \begin{empheq}[box = \fbox]{align*}
        \frac{T(t)}{T_0} = exp\left(-\frac{\alpha A}{M c_p}\right) = exp\left[-\frac{t}{\tau}\right], e^{-1} = 0.368
    \end{empheq}
\end{minipage}
\begin{minipage}{0.44\linewidth}
    \begin{scriptsize}
        \begin{empheq}{align*}
            A = \text{surface of lumped body}, \left[m^2\right]\\
            M = \text{mass of lumped body}, \left[kg\right]\\
            \tau = \text{time constant}, \left[s\right]
        \end{empheq}
    \end{scriptsize}
\end{minipage}
